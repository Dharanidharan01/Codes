let tasks = [];

function addTask(task) {
    tasks.push(task);
}

function showTasks() {
    console.log("To-Do List:");
    for (let i = 0; i < tasks.length; i++) {
        console.log((i + 1) + ". " + tasks[i]);
    }
}

function removeTask(task) {
    tasks = tasks.filter(t => t !== task);
}

// Example usage:
addTask("Buy shoes");
addTask("Clean store");
showTasks();
removeTask("Clean store");
showTasks();

todoApp.removeTask("Clean store");
todoApp.showTasks();
