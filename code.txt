// Class for each menu item
class MenuItem {
  constructor(Name, Price, Type) {
    this.Name = Name;
    this.Price = Price;
    this.Type = Type;
  }
}

// Class for the restaurant
class Restaurant {
  constructor() {
    this.menu = [];
  }

  // Add a new item to the menu
  addMenuItem(name, price, type) {
    const item = new MenuItem(name, price, type);
    this.menu.push(item);
  }

  // Show all menu items
  showMenu() {
    console.log("Restaurant Menu:");
    this.menu.forEach((item, index) => {
      console.log(`${index + 1}. ${item.Name} - ${item.Price} - (${item.Type})`);
    });
  }
}

// Example usage
const myRestaurant = new Restaurant();
myRestaurant.addMenuItem("Dosa", 250, "veg");
myRestaurant.addMenuItem("Chicken", 550, "non-veg");
myRestaurant.addMenuItem("Idly", 100, "veg");

myRestaurant.showMenu();
