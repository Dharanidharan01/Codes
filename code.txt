describe('Red Bull Event Page - Click Calendar Date Section', () => {
  it('Should visit the event page and click the calendar/date section', () => {
    // Step 1: Visit Red Bull India homepage
    cy.visit('https://www.redbull.com/in-en')

    // Step 2: Click on "Events"
    cy.get('a.header-navigation-links__item[href="/events"]', { timeout: 10000 })
      .should('be.visible')
      .click()

    // Step 3: Click on the specific event
    cy.get('img[alt="Red Bull King of the Air Qualifier Denmark Key Visual"]', { timeout: 10000 })
      .should('be.visible')
      .click()

    // Step 4: Click on the calendar/date section
    cy.get('.unified-event-hero-date-and-location__item-wrapper', { timeout: 10000 })
      .should('be.visible')
      .click()
  })
})
