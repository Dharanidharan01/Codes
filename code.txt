class Car {
  constructor() {
    this.isEngineOn = false;
  }

  start() {
    this.isEngineOn = true;
    console.log("Car started");
  }

  drive() {
    if (this.isEngineOn) {
      console.log("Car is driving");
    } else {
      console.log("Please start the car first");
    }
  }

  stop() {
    if (this.isEngineOn) {
      console.log("Car stopped");
      this.isEngineOn = false;
    } else {
      console.log("Car is already off");
    }
  }
}

// Using abstraction
const myCar = new Car();
myCar.start();
myCar.drive();
myCar.stop();
