// Sample salary calculation function
function calcSalary(base, bonus) {
  return base + bonus;
}
 
// Array of employees with base and bonus
let employees = [
  { name: "Arul", base: 40000, bonus: 5000 },
  { name: "Kumar", base: 45000, bonus: 7000 },
  { name: "Divya", base: 42000, bonus: 10000 },
  { name: "Ravi", base: 48000, bonus: 3000 }
];
 
// Add calculated salary to each employee
for (let i = 0; i < employees.length; i++) {
  employees[i].salary = calcSalary(employees[i].base, employees[i].bonus);
}
 
// Find highest earner
function findHighestEarner() {
  let highest = employees[0];
 
  for (let i = 1; i < employees.length; i++) {
    if (employees[i].salary > highest.salary) {
      highest = employees[i];
