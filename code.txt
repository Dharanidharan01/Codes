describe('DOB Selector - Dropdowns', () => {
  it('Selects DOB using select()', () => {
    cy.visit('https://demo.automationtesting.in/Register.html');
 
    cy.get('#year').select('1995');  // select year
    cy.get('#month').select('May');  // select month
    cy.get('#day').select('15');     // select day
  });
});
