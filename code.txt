let todoApp = {
    tasks: [],

    addTask(taskName) {
        this.tasks.push({ name: taskName, completed: false });
    },

    removeTask(taskName) {
        this.tasks = this.tasks.filter(task => task.name !== taskName);
    },

    showTasks() {
        this.tasks.forEach(task => {
            console.log(task.name + " - " + (task.completed ? "Done" : "Pending"));
        });
    }
};

// Example usage:
todoApp.addTask("Buy shoes");
todoApp.addTask("Clean store");
todoApp.showTasks();
todoApp.removeTask("Clean store");
todoApp.showTasks();
